{Cлияние двух упорядоченных последовательностей}
{Алгоритм решения  задачи  состоит  в  следующем:  сравниваем текущие элементы первой и второй последовательностей, первую обрабатываем начиная с начала, вторую - с конца до тех  пор,  пока  не закончиться  хотя  бы  одна  из  последовательностей,  меньший  из элементов помещая в новую последовательность, затем   дописываем в нее оставшиеся элементы.}
 var a,b,c:array [1..100] of integer;
     l,m,n,i,j:integer;
 begin
     write('Введите размерность 1-й последовательности ');
       readln(l);
     writeln('Введите элементы, упорядоченные по возрастанию');
       read(a[1]);
       for i:=2 to l do
       {проверка упорядоченности}
           repeat
              read(a[i]);
           until a[i-1] < a[i] ;
     write('Введите размерность 2-й последовательности ');
         readln(m);
     writeln('Введите элементы , упорядоченные по убыванию');
       read(b[1]);
       for i:=2 to m do
           repeat
              read(b[i]);
           until b[i-1] > b[i] ;
     i:=1; j:=m; n:=1;
     while (i<=l) and (j>=1) do
     {Обработка общей части последовательности}
       begin
           if a[i]<b[j] then
              begin
                  c[n]:=a[i];
                  i:=i+1;
                  n:=n+1;
              end
           else
             begin
                c[n]:=b[j];
                j:=j-1;
                n:=n+1;
             end;
     end;
     while (j>=1)  do
     {Обработка оставшихся элементов из В}
       begin
        c[n]:=b[j];
        j:=j-1;
        n:=n+1;
       end;
     while (i<=l) do
     {Обработка оставшихся элементов из A}
         begin
           c[n]:=a[i];
           i:=i+1;
           n:=n+1;
         end;
 writeln('Полученная последовательность имеет вид:');
    for i:=1 to n-1 do
      write(c[i]:6);
     end.
